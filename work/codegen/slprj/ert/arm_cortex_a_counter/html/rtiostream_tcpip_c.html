136/* Function: socketDataPending =================================================



137 * Abstract:



138 *  Returns true, via the 'pending' arg, if data is pending on the comm line.



139 *  Returns false otherwise.



140 *



141 *  RTIOSTREAM_NO_ERROR is returned on success, RTIOSTREAM_ERROR on failure.



142 */



143static int socketDataPending(



144    const SOCKET sock,



145    int    *outPending, 



146    unsigned int timeoutSecs)



147{



148    fd_set          ReadFds;



149    int             pending;



150    struct timeval  tval;



151    int retVal = RTIOSTREAM_NO_ERROR;



152    



153    FD_ZERO(&ReadFds);



154    FD_SET(sock, &ReadFds);



155



156    tval.tv_sec  = timeoutSecs;



157    tval.tv_usec = 0;



158



159    /*



160     * Casting the first arg to int removes warnings on windows 64-bit



161     * platform.  It is safe to cast a SOCKET to an int here because on



162     * linux SOCKET is typedef'd to int and on windows the first argument



163     * to select is ignored (so it doesn't matter what the value is).



164     */



165    pending = select((int)(sock + 1), &ReadFds, NULL, NULL, &tval);



166    if (pending == SOCK_ERR) {



167        retVal = RTIOSTREAM_ERROR;



168    }



169



170    *outPending = (pending==1);



171    return(retVal);    



172



173} /* end socketDataPending */

176/* Function: socketDataGet =====================================================



177 * Abstract:



178 *  Attempts to gets the specified number of bytes from the specified socket.



179 *  The number of bytes read is returned via the 'sizeRecvd' parameter.



180 *  RTIOSTREAM_NO_ERROR is returned on success, RTIOSTREAM_ERROR is returned on



181 *  failure.



182 *



183 * NOTES:



184 *  o it is not an error for 'sizeRecvd' to be returned as 0



185 *  o this function blocks if no data is available



186 */



187static int socketDataGet(



188    const SOCKET   sock,



189    char          *dst,



190    const size_t   size,



191    size_t        *sizeRecvd)



192{



193    int nRead;



194    int retVal;



195



196    nRead = recv(sock, dst, (int)size, 0U);



197



198    if (nRead == SOCK_ERR) {



199        retVal = RTIOSTREAM_ERROR;



200    } else {



201        retVal = RTIOSTREAM_NO_ERROR;



202    }



203



204    if (retVal!=RTIOSTREAM_ERROR) {



205        *sizeRecvd = (size_t) nRead;



206    }



207



208    return retVal;



209} /* end socketDataGet */

212/* Function: socketDataSet =====================================================



213 * Abstract:



214 *  Utility function to send data via the specified socket



215 */



216static int socketDataSet(



217    const SOCKET sock,



218    const void *src,



219    const size_t size,



220    size_t *sizeSent)



221{



222    int nSent;    



223    int sizeLim;



224    int retVal = RTIOSTREAM_NO_ERROR;



225    



226    /* Ensure size is not out of range for socket API send function */



227    if (size > (size_t) INT_MAX) {



228        sizeLim = INT_MAX;



229    } else {



230        sizeLim = (int) size;



231    }



232



233    nSent = send(sock, src, sizeLim, 0);



234



235    if (nSent == SOCK_ERR) {



236        retVal = RTIOSTREAM_ERROR;



237    } else { 



238        *sizeSent = (size_t)nSent;



239    }



240



241    return retVal;



242}

244/* Function: serverStreamRecv =================================================



245 * Abstract:



246 *  Send data from the server-side



247 */



248static int serverStreamRecv( 



249    void * dst,



250    size_t size,



251    size_t * sizeRecvd)



252{



253    int retVal = RTIOSTREAM_NO_ERROR;



254    int pending = 1;



255



256    *sizeRecvd = 0;



257



258    if (ServerData.sock == INVALID_SOCKET) {



259        



260        /* Attempt to open connection */



261        ServerData.sock = serverOpenConnection();



262    }   



263



264    if (ServerData.sock != INVALID_SOCKET) {



265        



266        if (Blocking==0) {



267            retVal = socketDataPending(ServerData.sock, &pending, 0);



268        }



269        



270        if ( (pending !=0) && (retVal==RTIOSTREAM_NO_ERROR) && (size>0) ) {



271            



272            retVal = socketDataGet(ServerData.sock, (char *)dst, size, sizeRecvd);



273            



274            if (*sizeRecvd == 0) { /* Connection closed gracefully by client */



275                



276                close(ServerData.sock);



277                ServerData.sock = INVALID_SOCKET;



278            }



279        }



280        



281        if ( retVal == RTIOSTREAM_ERROR ) {



282            



283            close(ServerData.sock);



284            ServerData.sock = INVALID_SOCKET;



285        }



286    }



287



288    return retVal;



289}

291/* Function: serverStreamOpen =================================================



292 * Abstract:



293 *  Opens the listening socket to be used for accepting an incoming connection.



294 */



295static int serverStreamOpen(void)



296{



297



298    struct sockaddr_in serverAddr;



299    int sockStatus;



300



301    int sFdAddSize     = (int) sizeof(struct sockaddr_in);



302



303    SOCKET lFd         = ServerData.listenSock;



304    int option         = 1;     



305    int streamID       = SERVER_STREAM_ID;



306



307    



308    if (lFd == INVALID_SOCKET) {



309        /*



310         * Create a TCP-based socket.



311         */



312        memset((void *) &serverAddr,0,(size_t)sFdAddSize);



313        serverAddr.sin_family      = AF_INET;



314        serverAddr.sin_port        = htons((unsigned short int) ServerData.port);



315        serverAddr.sin_addr.s_addr = htonl(INADDR_ANY);



316        



317        lFd = socket(AF_INET, SOCK_STREAM, 0);



318



319        if (lFd == INVALID_SOCKET) {



320            printf("socket() call failed.\n");



321        } else {



322            /*



323             * Listening socket should always use the SO_REUSEADDR option



324             * ("Unix Network Programming - Networking APIs:Sockets and XTI",



325             *   Volume 1, 2nd edition, by W. Richard Stevens).



326             */



327            sockStatus = 



328                setsockopt(lFd,SOL_SOCKET,SO_REUSEADDR,(char*)&option,sizeof(option));



329            if (sockStatus == SOCK_ERR) {



330                printf("setsocketopt() call failed.\n");



331                close(lFd);



332                lFd = INVALID_SOCKET;



333            }



334            /* Disable Nagle's Algorithm*/ 



335            sockStatus = 



336            setsockopt(lFd,IPPROTO_TCP,TCP_NODELAY,(char*)&option,sizeof(option));



337            if (sockStatus == SOCK_ERR) { 



338                printf("setsocketopt() TCP_NODELAY call failed.\n");



339                close(lFd); 



340                lFd = INVALID_SOCKET; 



341            } 



342        }



343        



344        if (lFd != INVALID_SOCKET) {



345            sockStatus = bind(lFd, (struct sockaddr *) &serverAddr, sFdAddSize);



346            if (sockStatus == SOCK_ERR) {



347                printf("bind() call failed: %s\n", strerror(errno));



348                close(lFd);



349                lFd = INVALID_SOCKET;



350            }



351        }



352        



353        if (lFd != INVALID_SOCKET) {



354            if (ServerData.port == 0) {



355               /* port 0 specifies dynamic free port allocation



356                * reuse serverAddr to store the actual address / port */



357               sockStatus = getsockname(lFd, (struct sockaddr *) &serverAddr, &sFdAddSize);           



358               if (sockStatus == SOCK_ERR) {



359                  fprintf(stderr,"getsockname() call failed: %s\n", strerror(errno));



360                  close(lFd);



361                  lFd = INVALID_SOCKET;               



362               }



363               else {                  



364                  /* write the server port number to stdout */



365                  SERVER_PORT_PRINTF("Server Port Number: %u\n", ntohs(serverAddr.sin_port));



366               }                 



367            }



368        }



369



370        if (lFd != INVALID_SOCKET) {



371            sockStatus = listen(lFd, 2);



372            if (sockStatus == SOCK_ERR) {



373                printf("listen() call failed.\n");



374                close(lFd);



375                lFd = INVALID_SOCKET;



376            }



377        }



378        ServerData.listenSock = lFd;



379    }



380



381    if (lFd == INVALID_SOCKET) {



382        streamID = RTIOSTREAM_ERROR;



383    }



384    return streamID;



385



386}



387/* Function: serverOpenConnection =================================================



388 * Abstract:



389 *  Called when the target is not currently connected to the host, this 



390 *  function attempts to open the connection.  



391 *



392 *  In the case of sockets, this is a passive operation in that the host



393 *  initiates contact, the target simply listens for connection requests.



394 *



395 * NOTES:



396 



397 * Blocks if Blocking == 1, poll for pending connections otherwise. When



398 * polling, there may be no open requests pending.  In this case, this



399 * function returns without making a connection; this is not an error.



400 */



401static SOCKET serverOpenConnection(void)



402{



403    struct sockaddr_in clientAddr;



404    int     sFdAddSize     = (int) sizeof(struct sockaddr_in);



405    SOCKET  cFd            = INVALID_SOCKET;



406    SOCKET  lFd            = ServerData.listenSock;



407    int error             = RTIOSTREAM_NO_ERROR;



408    int pending            = 1;



409



410    /* Check that the listening socket is still valid and open a new socket if



411     * not */



412    if (lFd == INVALID_SOCKET) {



413        serverStreamOpen();



414        lFd = ServerData.listenSock;



415    }



416



417    if (Blocking==0) {



418        error = socketDataPending(lFd, &pending, 0);



419    }



420    



421    



422    if ( (pending > 0) && (error==RTIOSTREAM_NO_ERROR) ) {



423



424        /*



425         * Wait to accept a connection on the comm socket.



426         */



427        cFd = accept(lFd, (struct sockaddr *)&clientAddr,



428                     &sFdAddSize);



429        



430        if (cFd == INVALID_SOCKET) {



431            printf("accept() for comm socket failed.\n");



432            error = RTIOSTREAM_ERROR;



433        } 



434



435        if (error == RTIOSTREAM_ERROR) {



436            close(ServerData.listenSock);



437            ServerData.listenSock = INVALID_SOCKET;



438        } 



439    }



440



441    return cFd;



442}

444/* Function: processArgs ====================================================



445 * Abstract:



446 *  Process the arguments specified by the user when opening the rtIOStream.



447 *      



448 *  If any unrecognized options are encountered, ignore them.



449 *



450 * Returns zero if successful or RTIOSTREAM_ERROR if 



451 * an error occurred.



452 *



453 *  o IMPORTANT!!!



454 *    As the arguments are processed, their strings should be NULL'd out in



455 *    the argv array. 



456 */



457static int processArgs(



458    const int      argc,



459    void         * argv[],



460    char        ** hostName, 



461    unsigned int * portNum,



462    unsigned int * isClient,



463    unsigned int * isBlocking,



464    unsigned int * recvTimeout)



465{



466    int        retVal    = RTIOSTREAM_NO_ERROR;



467    int        count           = 0;



468



469    while(count < argc) {



470        const char *option = (char *)argv[count];



471        count++;



472



473        if (option != NULL) {



474



475            if ((strcmp(option, "-hostname") == 0) && (count != argc)) {



476



477                *hostName = (char *)argv[count];



478                count++;



479                argv[count-2] = NULL;



480                argv[count-1] = NULL;



481



482            } else if ((strcmp(option, "-port") == 0) && (count != argc)) {



483                char       tmpstr[2];



484                int itemsConverted;



485                const char *portStr = (char *)argv[count];



486



487                count++;     



488                



489                itemsConverted = sscanf(portStr,"%d%1s", (int *) portNum, tmpstr);



490                if ( (itemsConverted != 1) || 



491                     ( ((*portNum != 0) && (*portNum < 255)) || (*portNum > 65535)) 



492                    ) {



493                    



494                    retVal = RTIOSTREAM_ERROR;



495                } else {



496



497                    argv[count-2] = NULL;



498                    argv[count-1] = NULL;



499                }           



500                



501            } else if ((strcmp(option, "-client") == 0) && (count != argc)) {



502                



503                *isClient = ( strcmp( (char *)argv[count], "1") == 0 );



504



505                count++;



506                argv[count-2] = NULL;



507                argv[count-1] = NULL;



508



509            } else if ((strcmp(option, "-blocking") == 0) && (count != argc)) {



510                



511                *isBlocking = ( strcmp( (char *)argv[count], "1") == 0 );



512



513                count++;



514                argv[count-2] = NULL;



515                argv[count-1] = NULL;



516



517            } else if ((strcmp(option, "-recv_timeout_secs") == 0) && (count != argc)) {



518                char       tmpstr[2];



519                int itemsConverted;



520                const char *timeoutSecsStr = (char *)argv[count];



521



522                count++;     



523                



524                itemsConverted = sscanf(timeoutSecsStr,"%d%1s", (int *) recvTimeout, tmpstr);



525                if ( itemsConverted != 1 ) {



526                    retVal = RTIOSTREAM_ERROR;



527                } else {



528



529                    argv[count-2] = NULL;



530                    argv[count-1] = NULL;



531                }           



532



533            } else {



534                /* do nothing */



535            }



536        }



537    }



538    return retVal;



539}

543/* Function: rtIOStreamOpen =================================================



544 * Abstract:



545 *  Open the connection with the target.



546 */



547int rtIOStreamOpen(int argc, void * argv[])



548{



549    char               *xHostName = NULL;



550    unsigned int        xPortNum     = (SERVER_PORT_NUM);



551    unsigned int        isClient = 0;



552    int result = RTIOSTREAM_NO_ERROR;



553    int streamID = RTIOSTREAM_ERROR;



554



555    Blocking = EXT_BLOCKING; /* Set default value */



556



557    processArgs(argc, argv, &xHostName, &xPortNum, &isClient, &Blocking,



558        &RecvTimeoutSecs);    



559



560    if (result != RTIOSTREAM_ERROR) {



561        if (isClient == 1) {



562



563        } else {



564            ServerData.port = xPortNum;



565            ServerData.sock = INVALID_SOCKET;



566            streamID = serverStreamOpen();



567        }



568    }



569    return streamID;



570}

572/* Function: rtIOStreamSend =====================================================



573 * Abstract:



574 *  Sends the specified number of bytes on the comm line. Returns the number of



575 *  bytes sent (if successful) or a negative value if an error occurred. As long



576 *  as an error does not occur, this function is guaranteed to set the requested



577 *  number of bytes; the function blocks if tcpip's send buffer doesn't have



578 *  room for all of the data to be sent



579 */



580int rtIOStreamSend(



581    int streamID,



582    const void *src,



583    size_t size,



584    size_t *sizeSent)



585{



586    int retVal;



587



588    if (streamID == SERVER_STREAM_ID) {



589        if (ServerData.sock == INVALID_SOCKET) {



590



591            ServerData.sock = serverOpenConnection();



592



593        }



594    



595        retVal = socketDataSet(ServerData.sock, src, size, sizeSent);



596



597    } else { /* Client stream */



598



599        SOCKET sock = (SOCKET) streamID;



600        retVal = socketDataSet(sock, src, size, sizeSent);



601



602    }



603    return retVal;



604}

607/* Function: rtIOStreamRecv ================================================



608 * Abstract: receive data



609 *



610 */



611int rtIOStreamRecv(



612    int      streamID,



613    void   * dst, 



614    size_t   size,



615    size_t * sizeRecvd) 



616{



617    int retVal = RTIOSTREAM_NO_ERROR;



618



619    if (streamID == SERVER_STREAM_ID) {



620



621        retVal = serverStreamRecv(dst, size, sizeRecvd); 



622



623    } else { /* Client stream */



624



625        int pending = 1;



626        SOCKET cSock = (SOCKET) streamID;



627



628        {



629            unsigned int timeout=0;



630            if (Blocking == 0) {



631                timeout = 0;



632            } else {



633                timeout = RecvTimeoutSecs;



634            }



635            retVal = socketDataPending(cSock,&pending,timeout);



636        }



637



638        if (pending==0) {



639            *sizeRecvd = 0U;



640        } else {



641            retVal = socketDataGet(cSock, (char *)dst, size, sizeRecvd);



642        }



643    }



644



645    return retVal;



646}

649/* Function: rtIOStreamClose ================================================



650 * Abstract: close the connection.



651 *



652 */



653int rtIOStreamClose(int streamID)



654{



655    int retVal = RTIOSTREAM_NO_ERROR;



656    



657    if (streamID == SERVER_STREAM_ID) {



658        char * tmpBuf[TMP_BUF_SIZ];



659        int numRecvd;



660        numRecvd = recv( ServerData.sock, (void *) tmpBuf, TMP_BUF_SIZ, 0);



661        while (numRecvd > 0) {



662            numRecvd = recv( ServerData.sock, (void *) tmpBuf, TMP_BUF_SIZ, 0);



663        }



664        close(ServerData.sock);



665        ServerData.sock = INVALID_SOCKET;



666



667        close(ServerData.listenSock);



668



669        ServerData.listenSock = INVALID_SOCKET;



670



671    } else {



672        SOCKET cSock = (SOCKET) streamID;



673        close(cSock);



674        



675    }



676



677    return retVal;



678}
